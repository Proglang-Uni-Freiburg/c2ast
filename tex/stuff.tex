  f &: \TFun\TInt{\TRec{\TIn \TInt X}}
  \\
  f &= \ELam x \EFork{(\ERec g \ELam {(c: \TRec{\TOut \TInt X})} g\; (\ESend x {(\EUnroll c)}))}
\end{align*}
